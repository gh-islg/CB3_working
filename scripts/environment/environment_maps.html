CB3 Equity Report — Environmental Conditions Maps

Purpose

This notebook reads the clean Environmental Conditions tract table and maps four representative dimensions:

  1. Rodent inspections (confirmed activity): rodent_active_inspections
  2. Street and sidewalk cleanliness: avg_acceptable_streets_pct, avg_acceptable_sidewalks_pct
  3. Street tree health: individual living street trees (2015 TreesCount census) colored by health status (Good/Fair/Poor)
  4. Hurricane evaluation zones: these are overlaid with NYCHA housing and subsidized units at risk
  5. EJ Area designation: ej_area_flag (shown as a hatched overlay on every map)
  6. Fine particulate matter (PM2.5): NYCCAS modeled annual average, Dec 2023-Dec 2024, shown as its native 300m grid cells rather than a tract statistic

Every numeric metric is presented as a bubble layer sized by magnitude over selectable demographic choropleth base layers. Rodent inspections are point events with native building coordinates, so their bubbles plot at each building’s own address. Street and sidewalk cleanliness are reported at the DSNY cleaning section level, not census tract, so their bubbles sit at each section’s centroid instead. Street tree health has no magnitude to size by — each tree is plotted at its own coordinates as a fixed-size marker colored by category instead of a bubble. Map HTML files are written to maps/environment/.

def show_map(output_path, height=650):
    """Display a saved map via a lightweight <iframe src=...> reference
    instead of returning the Folium object directly. Returning the object
    triggers Folium's _repr_html_, which inlines the ENTIRE map document as
    one escaped srcdoc attribute — for point-heavy maps (e.g. street tree
    health, ~6MB with 4,709 markers) that overflows Quarto's markdown
    formatter when it tries to process the cell output. Quarto always
    renders this .qmd's HTML report alongside the .qmd itself
    (scripts/environment/), so the relative path back up to maps/environment/
    is fixed regardless of the working directory execution used.
    """
    return IFrame(f"../../maps/environment/{output_path.name}", width="100%", height=height)

# Build METRIC_SPECS and DEMOGRAPHIC_SPECS from metric_metadata.yml.
# ej_area_flag is excluded here — it is rendered as a dedicated overlay, not a
# bubble layer, because it is binary and the concept doc calls for a hatched overlay.
def _to_map_spec(meta, include_filename=False):
    spec = {
        "dimension": meta["short_label"],
        "label":     meta["label"],
        "unit":      meta["unit"],
        "palette":   meta["palette"],
    }
    if include_filename:
        spec["filename"] = meta["filename"]
    for key in ("positive_values_only", "zero_color", "zero_label"):
        if key in meta:
            spec[key] = meta[key]
    return spec

# Tract-level environment metrics (excludes ej_area_flag, section-level
# metrics, and the non-tract layers below, which use category_colors instead
# of palette and are read directly by key rather than built into a spec dict).
METRIC_SPECS = {
    var: _to_map_spec(meta, include_filename=True)
    for var, meta in ALL_METRICS.items()
    if (
        meta["domain"] == "environment"
        and var != "ej_area_flag"
        and "filename" in meta
        and "palette" in meta
        and not meta.get("geography", "").startswith("DSNY")
    )
}

# Section-level sanitation metrics (DSNY cleaning sections, not census tracts).
SECTION_METRIC_SPECS = {
    var: _to_map_spec(meta, include_filename=True)
    for var, meta in ALL_METRICS.items()
    if meta.get("geography", "").startswith("DSNY") and "filename" in meta
}

# senior_poverty_rate is registered as a demographic metric for other domains'
# notebooks (e.g. seniors), but is not featured as a layer here.
EXCLUDED_DEMOGRAPHIC_METRICS = {"senior_poverty_rate"}

DEMOGRAPHIC_SPECS = {
    var: _to_map_spec(meta)
    for var, meta in ALL_METRICS.items()
    if meta["domain"] == "demographics" and var not in EXCLUDED_DEMOGRAPHIC_METRICS
}

for metric in METRIC_SPECS:

def add_ej_overlay(map_object, show=True):
    """Add EJ-designated tracts as a dashed-border checkbox overlay.
    """
    ej_tracts = tracts[tracts["ej_area_flag"].eq(1)][["GEOID", "tract_label", "nta_name", "geometry"]].copy()
    folium.GeoJson(
        data=ej_tracts.to_json(),
        name="EJ Area (2024)",
        overlay=True,
        show=show,
        style_function=lambda _: {
            "fillColor": "#d95f02",
            "color": "#d95f02",
            "weight": 2,
            "dashArray": "5 4",
            "fillOpacity": 0.20,
        },
        highlight_function=lambda _: {
            "weight": 3,
            "fillOpacity": 0.35,
        },
        tooltip=folium.GeoJsonTooltip(
            fields=["tract_label", "nta_name"],
            aliases=["Census tract", "NTA"],
            sticky=False,
        ),
    ).add_to(map_object)
    add_zero_value_legend(
        map_object,
        "#d95f02",
        "Designated EJ Area (MOCEJ 2024)",
    )


# Thin, environment-domain-specific wrapper around build_metric_bubble_map,
# so call sites below don't have to repeat tracts/DEMOGRAPHIC_SPECS/MAP_OUTPUT_DIR
# or re-add the EJ overlay on every call. Every other map-building function
# in this notebook (section, hurricane) routes through this one instead of
# calling build_metric_bubble_map directly, so the EJ overlay is added in
# exactly one place. metric_specs/title default to the tract-level METRIC_SPECS
# case; pass overrides for other geographies (DSNY sections) or ad hoc specs
# (the hurricane map's subsidized-housing bubble).
def create_single_metric_map(metric, metric_specs=None, title=None, extra_layers=None, **kwargs):
    metric_specs = metric_specs if metric_specs is not None else METRIC_SPECS
    spec = metric_specs[metric]

    # The EJ overlay is passed as build_metric_bubble_map's extra_layers hook
    # (composed with any caller-supplied extra_layers) so it's added before
    # that function's internal LayerControl, not after — a layer added after
    # LayerControl renders in a later script tag than the one that
    # references it, which throws a JS ReferenceError and silently breaks
    # the whole control.
    def combined_extra_layers(map_object):
        add_ej_overlay(map_object, show=False)
        if extra_layers is not None:
            extra_layers(map_object)

    return build_metric_bubble_map(
        tracts, metric_specs, DEMOGRAPHIC_SPECS, metric,
        title=title if title is not None else f"CB3 Environmental Conditions: {spec['dimension']}",
        output_path=MAP_OUTPUT_DIR / spec["filename"],
        extra_layers=combined_extra_layers,
        **kwargs,
    )

def add_section_outline_overlay(map_object, show=True):
    """Add DSNY cleaning section boundaries as an unfilled checkbox overlay.

    Bubbles sit at each section's centroid, so the outline gives viewers the
    actual section boundary the bubble is standing in for.
    """
    layer_data = sections_geometry[["section", "geometry"]]
    folium.GeoJson(
        data=layer_data.to_json(),
        name="DSNY cleaning sections",
        overlay=True,
        show=show,
        style_function=lambda _: {
            "fillOpacity": 0,
            "color": "#3825a5",
            "weight": 3,
        },
        tooltip=folium.GeoJsonTooltip(
            fields=["section"],
            aliases=["Cleaning section"],
            sticky=False,
        ),
    ).add_to(map_object)


def create_section_map(metric):
    return create_single_metric_map(

        tooltip_fields=["section"],
        tooltip_aliases=["Cleaning section"],

# Hurricane evacuation zones are already clipped to the CB3 tract boundary in
# environment.py (the raw citywide file is 7MB covering the whole coastline).
evac_zones_clipped = gpd.read_file(EVAC_ZONES_PATH)
evac_zones_clipped["evacuation_zone"] = evac_zones_clipped["evacuation_zone"].astype(str)

# Zone 1 evacuates first (highest risk); "X" means not in an evacuation zone.
# Colors and output filename are registered in metric_metadata.yml

def add_evacuation_zone_layer(map_object, show=True, legend_bottom_offset=35):
    """Add NYC Hurricane Evacuation Zones (clipped to CB3) as an always-on
    categorical polygon overlay, colored by zone severity. Unlike the
    demographic layers, zones aren't meant to be toggled one at a time —
    the map is read as one combined zone picture, same as NYC's own
    evacuation zone finder.
    """
    layer_data = evac_zones_clipped[["evacuation_zone", "geometry"]]
    folium.GeoJson(
        data=layer_data.to_json(),
        name="Hurricane evacuation zones",
        overlay=True,
        show=show,
        style_function=lambda feature: {
            "fillColor": EVAC_ZONE_COLORS.get(feature["properties"]["evacuation_zone"], "#d9d9d9"),
            "color": "#4d4d4d",
            "weight": 1,
            "fillOpacity": 0.55,
        },
        highlight_function=lambda _: {"weight": 3, "fillOpacity": 0.75},
        tooltip=folium.GeoJsonTooltip(
            fields=["evacuation_zone"],
            aliases=["Hurricane evacuation zone"],
            sticky=False,
        ),
    ).add_to(map_object)
    add_categorical_legend(
        map_object, "Hurricane evacuation zone", EVAC_ZONE_COLORS,
        bottom_offset=legend_bottom_offset, shape="square",
    )

# Ad hoc spec for create_single_metric_map: subsidized housing at risk is the
# one magnitude-sized layer on this map, so it doubles as the primary bubble
# metric; the evacuation zones (categorical polygons) and NYCHA buildings
# (no magnitude) are added via extra_layers instead of needing a bespoke map
# function. filename feeds create_single_metric_map's output_path.
HURRICANE_SUBSIDIZED_SPEC = {
    "dimension": "Subsidized housing at risk",
    "label": "Subsidized units at risk of expiring (2025-2040)",
    "unit": " units",
    "filename": HURRICANE_MAP_OUTPUT_PATH.name,
}


def create_hurricane_evacuation_map():
    # Legend offsets are staggered (35 / 260 / 460) so the subsidized-housing
    # bubble-size legend (up to 3 rows), the zone legend (up to 8 rows), and
    # the NYCHA point legend (1 row) stack without overlapping.
    def extra_layers(map_object):
        add_evacuation_zone_layer(map_object, show=True, legend_bottom_offset=260)
        add_categorical_point_layer(
            map_object,
            nycha_points,
            "category",
            NYCHA_COLORS,
            name="NYCHA residential buildings",
            tooltip_fields=["development_name", "address", "tract_label"],
            tooltip_aliases=["Development", "Address", "Census tract"],
            show=True,
            legend_bottom_offset=460,
        )

    return create_single_metric_map(
        "at_risk_subsidized_units",
        metric_specs={"at_risk_subsidized_units": HURRICANE_SUBSIDIZED_SPEC},
        title="CB3 Environmental Conditions: Hurricane Evacuation Zones & At-Risk Housing",
        points=subsidized_points,
        lat_col="latitude",
        lon_col="longitude",
        tooltip_fields=["standard_address", "tract_label"],
        tooltip_aliases=["Address", "Census tract"],
        color=SUBSIDIZED_BUBBLE_COLOR,
        extra_layers=extra_layers,
        subtitle=(
            "Choropleth: use the layer control to pick a demographic fill. Polygons: NYC hurricane evacuation "
            "zones (Zone 1 evacuates first). Bubbles: subsidized units with a known 2025-2040 expiration window "
            "(not the full subsidized stock). Points: NYCHA residential buildings."
        ),
    )

Fine particulate matter (PM2.5)

Code 
# NYCCAS PM2.5 is a modeled 300m-resolution surface, not a tract statistic
# (vectorized to one square polygon per grid cell in environment.py). Rather
# than drawing the raw squares, plot each cell at its own centroid as a bubble
# sized by magnitude, matching this notebook's "every numeric metric is a
# bubble" convention (see add_polygon_centroids, used the same way for the
# tract-level metrics and DSNY cleaning sections). Label/unit/filename are
# registered in metric_metadata.yml rather than hardcoded here.
PM25_SPEC = ALL_METRICS["pm25"]
PM25_OUTPUT_PATH = MAP_OUTPUT_DIR / PM25_SPEC["filename"]

pm25_grid = gpd.read_file(PM25_GRID_PATH)
pm25_points = add_polygon_centroids(
    pm25_grid, pm25_grid, "cell_id",
    lat_col="cell_centroid_latitude", lon_col="cell_centroid_longitude",
)


def create_pm25_map():
    map_object = make_base_map(tracts)
    add_demographic_backdrop_layers(map_object, tracts, DEMOGRAPHIC_SPECS)
    add_ej_overlay(map_object, show=False)
    add_bubble_layer(
        map_object,
        pm25_points,
        "pm25",
        PM25_SPEC["label"],
        PM25_SPEC["unit"],
        name="PM2.5 (NYCCAS, 300m grid)",
        lat_col="cell_centroid_latitude",
        lon_col="cell_centroid_longitude",
        tooltip_fields=[],
        tooltip_aliases=[],
        # PM2.5 varies only modestly (~6.5-10.6 ug/m3) and never approaches
        # zero, so the default zero-anchored scaling would bunch every bubble
        # near max_radius (see radius_for's docstring). Scale relative to the
        # observed min instead, so the least/most polluted cells actually
        # look different sizes.
